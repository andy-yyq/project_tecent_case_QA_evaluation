def mysql_query(**kwargs):
    '''
    通过指定sql查询语句（及动态参数，可以没有）查询结果
    :param sql_query: 查询语句
    :param kwarg: 动态参数
    :return:
    '''
    assert kwargs.get("sql_query"), f"入参缺失：sql_query "
    assert kwargs.get("mysql_config"), f"入参缺失：mysql_config "
    sql_query = kwargs["sql_query"]
    mysql_config = kwargs["mysql_config"]

    result = list()
    pattern = re.compile(r"\{.*?\}")
    while True:
        if pattern.search(sql_query):
            sign = pattern.search(sql_query).group()
            param = sign[1:-1]
            if kwargs.get(param):
                sql_query = re.sub(sign, kwargs[param], sql_query)
            else:
                raise CustomError(f"Mysql查询参数缺失或错误:{param}")
        else:
            break

    try:
        conn = pymysql.connect(**mysql_config)
        cursor = conn.cursor()
        cursor.execute(sql_query)

        size = 500
        while True:
            rows = cursor.fetchmany(size)
            if not rows:
                break
            result.extend(rows)
        cursor.close()
        conn.close()
    except Exception as e:
        msg = f"Error, mysql数据库查询失败a, {e}"
        return False, None, msg
    else:
        return True, result, ""

通过test_paper_verification.py页面实现如何功能：
1.读取表中所有数据作为列表，可以选择随机1条数据，也可以输入指定"问题ID"（对应表字段：question_id）获取到特定数据。无论哪种方式，可以通过"Next"按钮，切换到当前数据在列表中的下一条。
2.合理设计布局，分别展示"问题描述"（对应表字段：question）、"参考答案"（对应表字段：ground_truth）、"参考依据"（对应表字段：quote_info）,以及"源案例内容"。"源案例内容"是一个案例ID的选择列表，其中的案例ID需要从表字段：ori_answer的信息中抽取，ori_answer的信息是以字符串存储的json结构数据，结构如下所示），ori_answer字段数据结构实例：
{
    "final_answer": "本题答案...", 
    "quote_source": 
        {
            "16135996": "案例指出...", 
            "16087751": "案例指出...", 
            ...
        }
}
通过获取该结构中quote_source字段对应的dict数据，其中所有的key就是"源案例内容"列表中可选择案例ID。
当选择某个案例ID后，访问mysql表：v_all_ai_manual_case_info（该表的访问配置信息使用：settings.MYSQL_PROD_CONFIG），该表就2个字段:case_id和case_content,选择案例ID后，在页面上展示对应case_content的内容。如果找不到指定case_id，页面报错提示。

3.页面最后，增加一个当前内容的最终审核结论功能，可选择"通过"，"不通过"，并提交，一旦提交，则该条数据在casebase_qa_evaluation中final_check字段被修改为最终审核结论："通过"或"不通过"。
